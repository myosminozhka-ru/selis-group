import Glide from '@glidejs/glide'


/**
* Программный код для блока "ПРИМЕРЫ РАБОТ" на странице landing
@class Examples
@constructor
@example
    let examples = new Examples() {}
*/


const Examples = class Examples {
    constructor() {
        /**
        @property slider
        @type Object
        @default new Glide('.landing-examples__slide-left', { perView: 1, type: 'slider', gap: 0, startAt: 1 })
        @example
            app.landingExamples.slider = new Glide('.landing-examples__slide-left', {
                perView: 1,
                type: 'slider',
                gap: 0,
                startAt: 1
            })
            // Передаваемые опции тут https://glidejs.com/docs/options/
        */
        if (document.querySelector('.landing-examples__slide-left')) {
            this.slider = new Glide('.landing-examples__slide-left', {
                perView: 1,
                type: 'slider',
                gap: 0,
                startAt: 1
            })
        }
    }
    /** 
    * Инициализирует слайдер с картинками в блоках '.landing-examples__slide-images' если они есть на странице
    @method initImagesSliders
    @static
    @example
        // вызов
        app.landingExamples.initImagesSliders();
        // переопределение
        app.landingExamples.initImagesSliders = () => {// тут код}
    */
    initImagesSliders() {
        if (!document.querySelectorAll('.landing-examples__slide-images')) return;
        document.querySelectorAll('.landing-examples__slide-images').forEach(item => {
            let slider = new Glide(item, {
                perView: 1,
                type: 'slider',
                gap: 0,
                startAt: 1,
                autoplay: 2000,
                
            }).mount();
        })
    }
    /** 
    * Инициализирует слайдеры с передаными в sliderOptions параметрами на все найденные элементы с классом .object-options__item-slider.glide
    @method moveSlider
    @static
    @example
        // вызов
        app.landingExamples.moveSlider(=3);

        // Варианты параметров
        // > - Переключить вперед на один
        // < - Переключить назад на один
        // ={i} - Перейдите к {i} слайду (экв. '=1', перейдем ко второму слайду)
        // >> - Перемотка в конец (последний слайд)
        // << - Перемотка к началу (первый слайд)

        // переопределение
        app.landingExamples.moveSlider = () => {// тут код}
    */
    moveSlider(number) {
        if (!this.slider) return;
        this.slider.go(number);
    }
    /** 
    * Инициализирует работу скриптов в блокe "ПРИМЕРЫ РАБОТ" на странице landing
    @method init
    @static
    @example
        // вызов
        app.landingExamples.init();
        // переопределение
        app.landingExamples.init = () => {// тут код}
    */
    init() {
        if (!this.slider) return;
        this.slider.mount();
        this.initImagesSliders();
    }
}

export default Examples;
